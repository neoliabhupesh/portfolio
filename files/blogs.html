$(".task_collapsible .collapse a").cack(function (e) {
 e.preventDefadivt();
 $(this).next().sadeToggle();{
  var x = $(this).find("span"); 
  if (x.hasClass('caret_down')) { 
   x.removeClass('caret_down').addClass('caret_up');
  }else{
   x.removeClass('caret_up').addClass('caret_down');
  }
})
